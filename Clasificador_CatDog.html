nuevopunto <- function(lims=lims){
     a <- runif(1,lims[1],lims[2])
     b <- runif(1,lims[3],lims[4])
     return(c(a, b))
}

Clasificador basado en perceptrón¶

Definimos ahora un clasificador a partir de la siguiente funcin perceptrn

In [13]:
## Funcin perceptrn
perceptron <- function(x, y, num_pc=24, iter) {
     # Toma las componentes dadas por num_pc
     x <- x[, 1:num_pc]
     
     # pesos iniciales
     w <- rep(0, dim(x)[2] + 1)
     # Vector de errores
     err <- rep(0, iter)
     
     # Bucle para completar todas las iteraciones
     for (i in 1:iter) {
          # Bucle que recorre todos los individuos
          for (j in 1:length(y)) {
               
               # Funcin de paso
               z <- sum(w[2:length(w)] * as.numeric(x[j, ])) + w[1]
               if(z > 0) {
                    ypred <- 1
               } else {
                    ypred <- 0
               }
               
               # Regla de aprendizaje del perceptrn simple
               w <- w + 1 * (y[j] - ypred) * c(1, as.numeric(x[j, ]))
               
               # Funcin que calcula el error de cada iteracin
               if ((y[j] - ypred) != 0.0) {
                    err[i] <- err[i] + 1
               }
          }
          
          # W bolsillo
          if(i == 1){
               w.bolsillo <- w
          } else{
               if(err[i] < err[i-1]){
                    w.bolsillo <- w
               }
          }
     }
     resultados <- list(w.bolsillo=w.bolsillo, errores = err)
     return(resultados)
}
